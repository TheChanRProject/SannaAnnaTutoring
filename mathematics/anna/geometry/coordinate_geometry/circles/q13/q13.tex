\begin{center}
  \begin{align}
    2(x^2 + y^2 - 4x + 5y + 1) = 0 \\
    \frac{2(x^2 + y^2 - 4x + 5y + 1)}{2} = \frac{0}{2} \\
    x^2 + y^2 - 4x + 5y = -1 \\
    (x^2 - 4x + (\frac{-4}{2})^2) + (y^2 + 5y + (\frac{5}{2})^2) = -1 + 4 + \frac{25}{4} \\
    (x - 2)^2 + (y + \frac{5}{2})^2 = \frac{37}{4} \\
    \text{Center} = (2, -\frac{5}{2}) \\
    \text{Radius} = \sqrt{\frac{37}{4}}
  \end{align}
\end{center}
